-- EFF Leadership Academy: run once in Supabase SQL Editor.
-- Uses Supabase Auth for sign-in. This does not expose any secret keys.

create table if not exists public.academy_profiles (
  id uuid primary key references auth.users(id) on delete cascade,
  email text unique not null,
  full_name text,
  chapter text,
  position text,
  member_type text default 'Leader',
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.academy_progress (
  id bigint generated always as identity primary key,
  user_id uuid not null references auth.users(id) on delete cascade,
  course_id text not null,
  lesson_id text not null,
  completed_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  unique (user_id, course_id, lesson_id)
);

create table if not exists public.academy_lesson_engagement (
  id bigint generated always as identity primary key,
  user_id uuid not null references auth.users(id) on delete cascade,
  course_id text not null,
  lesson_id text not null,
  reflection text not null default '',
  quiz_passed boolean not null default false,
  updated_at timestamptz not null default now(),
  unique (user_id, course_id, lesson_id)
);

create table if not exists public.academy_submissions (
  id bigint generated always as identity primary key,
  user_id uuid not null references auth.users(id) on delete cascade,
  course_id text not null,
  lesson_id text not null,
  assignment_title text not null,
  response text not null,
  status text not null default 'Submitted',
  submitted_at timestamptz not null default now(),
  reviewed_at timestamptz,
  reviewer_note text,
  unique (user_id, course_id, lesson_id)
);

create table if not exists public.academy_staff (
  email text primary key,
  role text not null default 'Staff',
  added_at timestamptz not null default now()
);

create or replace function public.handle_academy_profile()
returns trigger language plpgsql security definer set search_path = public as $$
begin
  insert into public.academy_profiles (id, email, full_name, chapter, position)
  values (new.id, new.email, coalesce(new.raw_user_meta_data->>'full_name',''), coalesce(new.raw_user_meta_data->>'chapter',''), coalesce(new.raw_user_meta_data->>'position',''))
  on conflict (id) do update set email = excluded.email, full_name = coalesce(nullif(excluded.full_name,''), academy_profiles.full_name), chapter = coalesce(nullif(excluded.chapter,''), academy_profiles.chapter), position = coalesce(nullif(excluded.position,''), academy_profiles.position), updated_at = now();
  return new;
end;
$$;

drop trigger if exists on_academy_user_created on auth.users;
create trigger on_academy_user_created after insert on auth.users for each row execute procedure public.handle_academy_profile();

create policy "academy profiles: member update" on public.academy_profiles for update to authenticated using (auth.uid() = id) with check (auth.uid() = id);
